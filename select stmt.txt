REPORT ZAISWARYA_STRUCTURE2.
TYPES : BEGIN OF TY_ZAISWARYA_MAKT,
  EMP_ID TYPE ZAISWARYA_MAKT-EMP_ID,
  NAME TYPE ZAISWARYA_MAKT-NAME,
  SALARY TYPE ZAISWARYA_MAKT-SALARY,
  DEPARTMENT TYPE ZAISWARYA_MAKT-DEPARTMENT,
  END OF TY_ZAISWARYA_MAKT.

DATA : VARIABLE TYPE TY_ZAISWARYA_MAKT.
DATA v_desc TYPE ZAISWARYA_DEPT-DEPT_DESC.

PARAMETERS ID TYPE ZAISWARYA_MAKT-EMP_ID.

SELECT SINGLE EMP_ID NAME SALARY DEPARTMENT FROM ZAISWARYA_MAKT INTO VARIABLE WHERE EMP_ID EQ ID.
SELECT SINGLE DEPT_DESC FROM ZAISWARYA_DEPT INTO V_DESC WHERE dept_id EQ VARIABLE-DEPARTMENT.
  WRITE :/ VARIABLE-EMP_ID,
  VARIABLE-NAME,
  VARIABLE-SALARY,
  VARIABLE-DEPARTMENT,
  V_DESC.
*ENDSELECT.
